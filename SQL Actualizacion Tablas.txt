create table estadoPagoFacturasFirmadas(
   idEstadoPagoFacturasFirmadas INT NOT NULL AUTO_INCREMENT,
   obra VARCHAR(50),
   nroEstadoPagoAsociado INT,
   nroFactura int,
   adjunto VARCHAR(50),
   usuario VARCHAR(50),
   fecha datetime,
   PRIMARY KEY (idEstadoPagoFacturasFirmadas)
);
